122  void insert(const char* str, int tok) {
123  if (*str == '\0') {
124  AT_ASSERT(kind == 0);
125  kind = tok;
126  return;

128 
129  for (size_t i = 0, e = child_chars.size(); i < e; ++i) {

200  // python concatenates all adjacent strings "a" "b" == "ab"
201  // strings can be enclosed with 1 or 3 single or double quotes
202  // if enclosed with 3 quotes newlines are valid
203  // as elsewhere, backslash and new line should be ignored
204  bool isString(const std::string& str, size_t start, size_t* len) {
205  char quote = str[start];
206  if (quote != '\"' && quote != '\'')
207  return false;
208  int quote_len = isCharCount(quote, str, start, 3) ? 3 : 1;
209 
210  // end is now set past the opening quotation marks
211  size_t end = start + quote_len;
212  while (end < str.size() && !isCharCount(quote, str, end, quote_len)) {
213  if (str[end] == '\n' && quote_len != 3) {
214  return false;
215  }
216  // handle escaped characters. advances past escaped quotation marks,
217  // escaped newlines and escaped backslashes
218  // multi-char escapes like \x1A are handled fine here because the
219  // remainder of the escape are valid string characters anyway
220  if (str[end] == '\\') {
221  end++;
222  }
223  end++;
224  }
225  // set length equal to the complete string including quotations
226  *len = end - start + quote_len;
227  // if end finished without going past the last character of the string than
228  // there is a match
229  return end < str.size();
230  }

244  // find the longest match of str.substring(pos) against a token, return true
245  // if successful filling in kind, start,and len
246  bool match(
247  const std::string& str,
248  size_t pos,
249  bool continuation, // are we inside a scope where newlines don't count
250  // (e.g. inside parens)
251  bool whitespace_token, // should we treat whitespace as a token
252  int* kind,
253  size_t* start,
254  size_t* len) {
255  *start = pos;
256  // skip whitespace
257  while (pos < str.size() && isblank(str[pos]))
258  pos++;
259 
260  // special handling
261  if (pos < str.size()) {
262  if (str[pos] == '#' && !isTypeComment(str, pos)) {
263  // skip comments
264  while (pos < str.size() && str[pos] != '\n')
265  pos++;
266  // tail call, handle whitespace and more comments
267  return match(
268  str, pos, continuation, whitespace_token, kind, start, len);
269  }
270  if (str[pos] == '\\' && pos + 1 < str.size() && str[pos + 1] == '\n' &&
271  !whitespace_token) {
272  return match(str, pos + 2, continuation, false, kind, start, len);
273  }
274  if (str[pos] == '\n') {
275  return match(
276  str, pos + 1, continuation, !continuation, kind, start, len);
277  }
278  }
279  // we handle white space before EOF because in the case we have something
280  // like the following where we need to generate the dedent token if foo:
281  // ...
282  // else:
283  // pass
284  if (whitespace_token) {
285  *kind = pos == str.size() ? TK_WHITESPACE_EOF : TK_WHITESPACE;
286  *len = pos - *start;
287  return true;
288  }
289  if (pos == str.size()) {
290  *kind = TK_EOF;
291  *start = pos;
292  *len = 0;
293  return true;
294  }
295  // invariant: the next token is not whitespace or newline
296  *start = pos;
297  // check for a valid number
298  if (isNumber(str, pos, len)) {
299  *kind = TK_NUMBER;
300  return true;
301  }
302  // check for string
303  if (isString(str, pos, len)) {
304  *kind = TK_STRINGLITERAL;
305  return true;
306  }
307 
308  // check for either an ident or a token
309  // ident tracks whether what we have scanned so far could be an identifier
310  // matched indicates if we have found any match.
311  bool matched = false;
312  bool ident = true;
313  TokenTrie* cur = head.get();
314  for (size_t i = 0; pos + i < str.size() && (ident || cur != nullptr); i++) {
315  ident = ident && validIdent(i, str[pos + i]);
316  if (ident) {
317  matched = true;
318  *len = i + 1;
319  *kind = TK_IDENT;
320  }
321  // check for token second, so that e.g. 'max' matches the token TK_MAX
322  // rather the
323  // identifier 'max'
324  if (cur) {
325  size_t child_offset = 0;
326  for (size_t e = cur->child_chars.size(); child_offset < e;
327  ++child_offset) {
328  if (cur->child_chars[child_offset] == str[pos + i])
329  break;
330  }
331 
332  cur = (child_offset == cur->child_chars.size())
333  ? nullptr
334  : cur->child_tries[child_offset].get();
335 
336  if (cur && cur->kind != 0) {
337  matched = true;
338  *len = i + 1;
339  *kind = cur->kind;
340  }
341  }
342  }
343  return matched;
344  }

400  }

446  private:
447  void lex() {
448  auto r = lexRaw();
449  switch (r.kind) {
450  case '(':
451  case '[':
452  case '{':
453  nesting++;
454  break;
455  case ')':
456  case ']':
457  case '}':
458  nesting--;
459  break;
460  case TK_WHITESPACE:
461  case TK_WHITESPACE_EOF: {
462  int depth =
463  r.kind == TK_WHITESPACE_EOF ? indent_stack.front() : r.range.size();
464  // note: TK_WHITESPACE_EOF is whitespace right before the EOF token
465  // just like we allow the code to be indented to a particular initial
466  // indent level, we allow the final indent to be anything and set
467  // it back to the initial indent level. This allows the code to be
468  // put into string literals inside code without worrying about final
469  // whitespace
470  if (depth > indent_stack.back()) {
471  indent_stack.push_back(depth);
472  r.kind = TK_INDENT;
473  } else if (depth == indent_stack.back()) {
474  r.kind = TK_NEWLINE;
475  } else {
476  next_tokens.emplace_back(TK_NEWLINE, r.range);
477  while (indent_stack.back() != depth) {
478  indent_stack.pop_back();
479  next_tokens.emplace_back(TK_DEDENT, r.range);
480  if (indent_stack.size() == 0) {
481  reportError("invalid indent level " + std::to_string(depth), r);
482  }
483  }
484  return; // We've already queued the tokens
485  }
486  } break;
487  default:
488  break;
489  }
490  next_tokens.push_back(std::move(r));
491  }
492  Token lexRaw(bool whitespace_token = false) {
493  int kind;
494  size_t start;
495  size_t length;
496  AT_ASSERT(file);
497  if (!shared.match(
498  *file,
499  pos,
500  nesting > 0,
501  whitespace_token,
502  &kind,
503  &start,
504  &length)) {
505  expected(
506  "a valid token",
507  Token((*file)[start], SourceRange(file, start, start + 1)));
508  }
509  auto t = Token(kind, SourceRange(file, start, start + length));
510  pos = start + length;
511  return t;
512  }
