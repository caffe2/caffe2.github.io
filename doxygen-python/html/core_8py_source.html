273  """

424 class IR(object):
425  """A simple IR class to keep track of all intermediate representations used
426  in the gradient computation.
427  """
428 
429  def __init__(self, operators):
430  # The IR class holds multiple metadata from the forward pass:
431  # a) ssa: a list of [op, in_versions, out_versions] recording the
432  # input and the output version of each operator, similar
433  # to a normal SSA form.
434  # b) input_count: a dictionary specifying for each blob and
435  # each of its version, how many times it is used as input for another
436  # op.
437  # c) frontier: maintaining the current versions of the blobs
438  # we are having in the workspace, after the execution of all the ops
439  # added to the IR so far. This is useful because if a gradient is
440  # trying to access an earlier version of a blob, we can sanity check
441  # that it is no longer there, and thus throw an error.
442  # d) gradient_frontier: maps the names of blobs to its version that the
443  # gradient corresponds to.
444  # e) gradient_generators: for each blob and each of its version, maps to
445  # a list of operators that generates its gradient together with the
446  # gradient name.
447  self.ssa = []
448  self.input_usages = defaultdict(lambda: defaultdict(list))
449  self.frontier = defaultdict(int)
450  self.gradient_frontier = {}
451  self.gradient_generators = defaultdict(lambda: defaultdict(list))
452  self.out_version_history = defaultdict(list)
453  self.in_version_history = defaultdict(list)
454 
455  for op in operators:
456  self.Play(op)
457 
458  self.SanityCheck(operators)

470  def Play(self, op):
471  """"Adds an op to the current IR, and update the internal states to
472  reflect the blobs and versions after the execution of the op.
473  """
474  # For input, they are the current version in the dict.
475  in_versions = {}
476  for s in op.input:
477  in_versions[s] = self.frontier[s]
478  self.input_usages[s][self.frontier[s]].append(len(self.ssa))
479  self.in_version_history[s].append((op, self.frontier[s]))
480  # For output, they are the current version plus one. If this is a
481  # newly created blob, its version starts with zero.
482  out_versions = {}
483  for s in op.output:
484  if s in self.frontier:
485  self.frontier[s] += 1
486  out_versions[s] = self.frontier[s]
487  self.out_version_history[s].append((op, self.frontier[s]))
488  # Add to SSA for bookkeeping.
489  self.ssa.append(OpSSA(op, in_versions, out_versions))

585  def BuildGradientGenerators( # NOQA
586  self, fwd_op_idx, gradient_ops, g_output, g_input):
587  """Updates gradient_generators and gradient_frontier"""
588  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
589  locally_generated_blobs = []
590  sparse_generators = defaultdict(lambda: defaultdict(list))
591 
592  for grad_op in gradient_ops:
593  # (1) check that inputs are valid
594  for s in grad_op.input:
595  self.CheckGradientOperatorInput(
596  s, g_output, fwd_op_idx, locally_generated_blobs)
597 
598  # (2) add outputs to the locally generated blobs
599  # If an output corresponds to the gradient of an input, we also
600  # record it to gradient_generators
601  locally_generated_blobs.extend([str(s) for s in grad_op.output])
602  for i, output in enumerate(grad_op.output):
603  input_index = GetIndexFromGradientList(g_input, output)
604  if input_index is not None:
605  input_name = forward_op.input[input_index]
606  input_version = in_versions[input_name]
607  g = g_input[input_index]
608  if type(g) is GradientSlice:
609  # the output corresponds either to the indices or the
610  # values of the sparse gradient. In either case we
611  # create a (partial) SparseGradGenMeta. If necessary,
612  # we'll merge indices and values generators
613  # corresponding to the same gradient in step (3)
614  if g.indices == output:
615  m = SparseGradGenMeta(grad_op, i, None, 0, g)
616  else:
617  assert(g.values == output)
618  m = SparseGradGenMeta(None, 0, grad_op, i, g)
619  sparse_generators[input_name][input_version].append(m)
620  else:
621  self.gradient_generators[input_name][input_version] \
622  .append(GradGenMeta(
623  grad_op, i, g))
624 
625  # (3) merge indices and values generators for sparse gradients, and
626  # add them to gradient_generators
627  self.AppendSparseGenerators(sparse_generators)
628 
629  # (4) for ops (e.g., Add, Sum, Sub) which have gradient outputs directly

840  def DoGradientAccumulation(self, fwd_op_idx):
841  """For each input name in the forward op, check if we will need to
842  add gradient accumulation. If so, do gradient accumulation and return
843  the list of gradient operators.
844 
845  The criteria for doing gradient accumulation is:
846  (1) the specific input version has been used by multiple operators.
847  (2) the current fwd_op_idx is the first to use that input, i.e. in the
848  backward pass, is the last to optionally generate the gradient for
849  the op.
850  (3) For the operators that used the input, their gradient operators
851  have generated more than 1 gradient.
852 
853  When accumulating operators, our current solution is to rename all the
854  created gradients with an internal intermediate name, and then add a
855  Sum() operator that adds up all the gradients. This may use more memory
856  due to intermediate storage, but is usually the fastest approach as one
857  can do one single sum for multiple intermediate gradients.
858  """
859  forward_op, in_versions, out_versions = self.ssa[fwd_op_idx]
860  additional_sum_ops = []
861  grad_map = {}
862  for _i, input_name in enumerate(set(forward_op.input)):
863  input_version = in_versions[input_name]
864  input_usage = self.input_usages[input_name][input_version]
865  if (len(input_usage) <= 1 or fwd_op_idx != input_usage[0]):
866  # We do not need to do gradient accumulation yet.
867  continue
868  generator = self.gradient_generators[input_name][input_version]
869  try:
870  if not self._VerifyGradientGenerators(generator):
871  continue
872  except RuntimeError as err:
873  raise RuntimeError(
874  "Gradients for param ''{}'' failed to verify: {}".format(
875  input_name,
876  err
877  )
878  )
879 
880  # Finally, let's create the sum operator.
881  sum_ops, g = self._MakeSumOps(input_name, input_version)
882  additional_sum_ops.extend(sum_ops)
883  grad_map[input_name] = g
884  return additional_sum_ops, grad_map

950  def GetBackwardPass(self, ys):
951  """Gets the backward pass that computes the derivatives of given blobs.
952 
953  Inputs:
954  ys: a list or a dictionary specifying what blobs we want to compute
955  derivatives of. If the input is a list, we will automatically
956  generate their gradients with all-one values; if the input is a
957  dictionary, for any dictionary entries that are not None, we will
958  take the corresponding blobs as their gradients; for all those
959  that are None, we will auto-fill them with 1.
960  """
961  if isinstance(ys, list):
962  ys = dict((y, None) for y in ys)
963  elif not isinstance(ys, dict):
964  raise TypeError("ys should either be a list or a dict.")
965 
966  # Set the gradient frontier with the initialized external
967  # gradients.
968  for y in viewkeys(ys):
969  self.gradient_frontier[y] = self.frontier[y]
970  self.input_usages[str(y)][self.frontier[str(y)]].append(
971  len(self.ssa))
972 
973  all_input_to_grad, all_gradient_ops = self._GetInitGradients(ys)
974 
975  # (2) Now, after having the virtual play above, we now play the ops
976  # backwards, creating the gradients along the path. Note that although
977  # we are playing it backwards, we cannot refer to variables that are
978  # at a version older than current_versions because it is already been
979  # overwritten.
980  for forward_op_idx in reversed(range(len(self.ssa))):
981  input_to_grad, gradient_ops = self._GenerateGradientsForForwardOp(
982  forward_op_idx, all_input_to_grad)
983  all_input_to_grad.update(input_to_grad)
984  all_gradient_ops += gradient_ops
985 
986  # If there are multiple use blobs, do gradient accumulation.
987  additional_sum_ops, grad_map = self.DoGradientAccumulation(
988  forward_op_idx)
989  # This line is so that if in an accumulation some of the operators
990  # have not produced gradients, they still do not overwrite the
991  # general all_input_to_grad map.
992  all_input_to_grad.update(grad_map)
993  all_gradient_ops += additional_sum_ops
994 
995  # (3) Post-processing.
996  # After we have done computation for each op, we now have the gradient
997  # operators ready. For the output map, we will convert everything to
998  # BlobReferences for easier handling in python.
999  all_input_to_grad_out = {}
1000  for key, val in viewitems(all_input_to_grad):
1001  if val is not None:
1002  if (isinstance(val, string_types) or
1003  isinstance(val, binary_type)):
1004  grad_out = BlobReference(val)
1005  else:
1006  grad_out = GradientSlice(BlobReference(val[0]),
1007  BlobReference(val[1]))
1008  all_input_to_grad_out[BlobReference(key)] = grad_out
1009  return all_gradient_ops, all_input_to_grad_out

1116  """
1117  Given a net, return a structure containing the version of each input and
1118  output blob used by each operator.
1119 
1120  Args:
1121  net: either a Net or a NetDef
1122  blob_versions: (optional) map with current version number for given
1123  blob names. If not provided or blob not found, start
1124  from version 0.
1125  Returns:
1126  Tuple (ssa, blob_versions)
1127  ssa: list of tuples (versioned_inputs, versioned_outputs)
1128  for each op in the net. A versioned input is a tuple
1129  (blob_name, version).
1130  blob_versions: updated map with latest version of each blob found in
1131  the net.
1132  """
1133  proto = net.Proto() if isinstance(net, Net) else net
1134  assert isinstance(proto, caffe2_pb2.NetDef)
1135  if blob_versions is None:
1136  blob_versions = {}
1137  if isinstance(net, list):
1138  return [get_ssa(n, blob_versions) for n in net], blob_versions
1139  for i in proto.external_input:
1140  if i not in blob_versions:
1141  blob_versions[str(i)] = 0

1144  if not proto.external_input:
1145  for i in op.input:
1146  if i not in blob_versions:
1147  blob_versions[i] = 0

1149  for o in op.output:
1150  blob_versions[str(o)] = blob_versions.get(str(o), 0) + 1
